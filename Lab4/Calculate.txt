package com.example;

public class Calculate {

    private double module;
    private double phase;

    public Calculate()
    {
        this.module=this.phase=0;
    }

    public Calculate(double real, double complex) {
        to_indicative(real, complex);
    }

   public void to_indicative(double real, double complex) {
       this.module = Math.sqrt(real*real+complex*complex);
       if(real!=0)
           this.phase = Math.atan(complex/real);
       else
           this.phase=1.57;
    }

    public double getmodule()
    {
        return this.module;
    }

    public double getphase()
    {
        return this.phase;
    }
}
