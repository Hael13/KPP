package com.example;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import java.util.HashMap;

@Component
public class Cache {

    private HashMap<String, Calculate> cache;

    public Cache() {
        this.cache=new HashMap<String, Calculate>();
    }

    public void add(String key, Calculate temp) {
        this.cache.put(key, temp);
    }

    public Calculate find(String real, String compl) {
        String key = real+ " "+compl;
        Calculate finded;
        finded = this.cache.get(key);
        return finded;
    }

}
